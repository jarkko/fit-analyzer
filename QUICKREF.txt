📥 SYNC FROM GARMIN CONNECT
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  python garmin_sync.py

  → Downloads new activities from Garmin Connect
  → Automatically analyzes all FIT files
  → Updates workout_summary_from_fit.csv
  → Avoids duplicates automatically ✨

📊 ANALYZE EXISTING FIT FILES
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  python fit_to_summary.py --ftp 300 --multisport *.fit

  → Processes all FIT files in current directory
  → Separates multisport activity sessions
  → Creates workout_summary_from_fit.csv

⚙️  COMMON OPTIONS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  --ftp 280              Your Functional Threshold Power
  --hrmax 185            Your maximum heart rate
  --hrrest 55            Your resting heart rate
  --days 7               Download last 7 days (sync only)
  --download-only        Download without analyzing
  --analyze-only         Analyze without downloading
